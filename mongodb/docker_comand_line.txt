docker run -d \
  --name mongodb \
  --restart on-failure:3 \
  --network wpp-network \
  -e MONGO_INITDB_DATABASE=$(grep MONGODB_DATABASE .env | cut -d '=' -f2) \
  -e MONGO_INITDB_ROOT_USERNAME=$(grep MONGODB_USER .env | cut -d '=' -f2) \
  -e MONGO_INITDB_ROOT_PASSWORD=$(grep MONGODB_PASSWORD .env | cut -d '=' -f2) \
  -v mongo_data:/data/db \
  -p 27017:27017 \
  mongo:latest
